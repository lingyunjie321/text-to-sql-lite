# 运行时数据库与模型路由配置 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现前端可配置数据库连接和 `light/strong` 两档模型路由，后端通过 `runtime_config_id` 在单次会话中使用对应数据库与模型配置。

**Architecture:** 新增 `runtime` 子模块负责运行时配置模型、内存存储、解析和连通性测试。`TextToSQLApiService` 按请求解析默认配置或运行时配置，再把数据库 URL、`RoutingLLMClient` 和 `light/strong` 模型 profiles 注入现有 workflow；`WorkflowEngine`、`NodeFactory`、`NodeRegistry` 不改变职责。

**Tech Stack:** Python 3.11+、FastAPI、Pydantic、SQLAlchemy、SQLGlot、pytest、ruff、React 18、TypeScript、Vitest。

```python
from __future__ import annotations

from datetime import UTC, datetime

from text_to_sql_demo.runtime.models import RuntimeConfig


class RuntimeConfigStore:
    """演示用内存运行时配置存储。"""

    def __init__(self) -> None:
        self._configs: dict[str, RuntimeConfig] = {}

    def save(self, config: RuntimeConfig) -> None:
        """保存运行时配置。"""
        self._configs[config.id] = config

    def get(self, config_id: str, *, now: datetime | None = None) -> RuntimeConfig | None:
        """读取未过期配置。"""
        config = self._configs.get(config_id)
        if config is None:
            return None
        current_time = now or datetime.now(UTC)
        if config.expires_at <= current_time:
            return None
        return config

    def get_raw(self, config_id: str) -> RuntimeConfig | None:
        """读取配置原始值，测试和诊断使用。"""
        return self._configs.get(config_id)

    def prune_expired(self, *, now: datetime | None = None) -> int:
        """删除已过期配置并返回删除数量。"""
        current_time = now or datetime.now(UTC)
        expired_ids = [
            config_id
            for config_id, config in self._configs.items()
            if config.expires_at <= current_time
        ]
        for config_id in expired_ids:
            del self._configs[config_id]
        return len(expired_ids)
```

```python
from text_to_sql_demo.llm.client import LLMClient, LLMResponse


class RoutingLLMClient:
    """按 model_alias 转发到不同 LLM client。"""

    def __init__(self, *, clients_by_alias: dict[str, LLMClient]) -> None:
        self.clients_by_alias = dict(clients_by_alias)

    def complete(
        self,
        *,
        model_alias: str,
        model_name: str,
        system_prompt: str,
        user_prompt: str,
        temperature: float = 0.0,
        max_tokens: int | None = None,
    ) -> LLMResponse:
        """把请求转发给 alias 对应的底层 client。"""
        client = self.clients_by_alias[model_alias]
        return client.complete(
            model_alias=model_alias,
            model_name=model_name,
            system_prompt=system_prompt,
            user_prompt=user_prompt,
            temperature=temperature,
            max_tokens=max_tokens,
        )
```

```python
from __future__ import annotations

from collections.abc import Callable
from dataclasses import dataclass
from datetime import UTC, datetime

from text_to_sql_demo.config.models import WorkflowConfig
from text_to_sql_demo.llm.client import LLMClient, MockLLMClient
from text_to_sql_demo.llm.models import ModelProfile
from text_to_sql_demo.llm.providers import OpenAICompatibleLLMClient
from text_to_sql_demo.llm.routing import RoutingLLMClient
from text_to_sql_demo.runtime.exceptions import (
    RuntimeConfigExpiredError,
    RuntimeConfigNotFoundError,
    RuntimeProviderUnsupportedError,
    RuntimeSecretMissingError,
)
from text_to_sql_demo.runtime.models import RuntimeDriver, RuntimeModelConfig
from text_to_sql_demo.runtime.store import RuntimeConfigStore
from text_to_sql_demo.sql.dialect import DialectName

DRIVER_TO_DIALECT: dict[str, DialectName] = {
    "sqlite": "sqlite",
    "postgresql": "postgres",
    "mysql": "mysql",
}


@dataclass(frozen=True, slots=True)
class ResolvedRuntimeConfig:
    """一次请求实际使用的数据库和模型依赖。"""

    database_url: str
    target_dialect: DialectName
    llm_client: LLMClient
    model_profiles: dict[str, ModelProfile]
    runtime_config_id: str | None = None


def driver_to_dialect(driver: RuntimeDriver) -> DialectName:
    """把数据库 driver 映射为 SQL 方言名称。"""
    return DRIVER_TO_DIALECT[driver]


class RuntimeConfigResolver:
    """把默认配置或运行时配置解析为 workflow 可注入依赖。"""

    def __init__(
        self,
        *,
        workflow_config: WorkflowConfig,
        store: RuntimeConfigStore,
        default_database_url: str,
        default_llm_client: LLMClient,
        now_provider: Callable[[], datetime] | None = None,
    ) -> None:
        self.workflow_config = workflow_config
        self.store = store
        self.default_database_url = default_database_url
        self.default_llm_client = default_llm_client
        self.now_provider = now_provider or (lambda: datetime.now(UTC))

    def resolve(self, *, runtime_config_id: str | None) -> ResolvedRuntimeConfig:
        """解析本次请求的运行时配置。"""
        if runtime_config_id is None:
            return ResolvedRuntimeConfig(
                database_url=self.default_database_url,
                target_dialect="sqlite",
                llm_client=self.default_llm_client,
                model_profiles=_model_profiles_from_workflow(self.workflow_config),
            )

        config = self.store.get_raw(runtime_config_id)
        if config is None:
            raise RuntimeConfigNotFoundError("运行时配置不存在")

        now = self.now_provider()
        if config.expires_at <= now:
            raise RuntimeConfigExpiredError("运行时配置已过期")

        model_configs = {
            "light": config.models.light,
            "strong": config.models.strong,
        }
        return ResolvedRuntimeConfig(
            database_url=config.database.database_url.get_secret_value(),
            target_dialect=config.database.target_dialect,
            llm_client=RoutingLLMClient(
                clients_by_alias={
                    alias: _build_llm_client(model_config)
                    for alias, model_config in model_configs.items()
                }
            ),
            model_profiles={
                alias: ModelProfile(
                    alias=alias,
                    provider=model_config.provider,
                    model_name=model_config.model,
                    temperature=model_config.temperature,
                    max_tokens=model_config.max_tokens,
                )
                for alias, model_config in model_configs.items()
            },
            runtime_config_id=runtime_config_id,
        )


def _model_profiles_from_workflow(config: WorkflowConfig) -> dict[str, ModelProfile]:
    """从 workflow 配置生成 light/strong 模型 profiles。"""
    return {
        alias: ModelProfile(
            alias=alias,
            provider=model_config.provider,
            model_name=model_config.model,
            temperature=model_config.temperature,
            max_tokens=model_config.max_tokens,
        )
        for alias, model_config in config.models.aliases.items()
    }


def _build_llm_client(config: RuntimeModelConfig) -> LLMClient:
    """根据运行时模型配置构造 provider client。"""
    if config.provider == "mock":
        return MockLLMClient()

    if config.provider == "openai_compatible":
        if config.api_key is None:
            raise RuntimeSecretMissingError("运行时模型缺少 API Key")
        return OpenAICompatibleLLMClient(
            api_key=config.api_key.get_secret_value(),
            base_url=config.base_url or "https://api.openai.com/v1/chat/completions",
        )

    raise RuntimeProviderUnsupportedError(f"不支持的运行时 provider: {config.provider}")
```

```tsx
import type { RuntimeOptionsResponse } from "../api/types";

interface RuntimeConfigPanelProps {
  options: RuntimeOptionsResponse;
  isSaving: boolean;
  onClose: () => void;

export function RuntimeConfigPanel({
  options,
  isSaving,
  onClose,
  onCreate
}: RuntimeConfigPanelProps): JSX.Element {
  const databasePreset = options.database_presets[0];
  const lightPreset = options.model_presets.light[0];
  const strongPreset = options.model_presets.strong[0];

  function submit(): void {
    onCreate({
      database: { mode: "preset", preset_id: databasePreset.id },
      models: {
        light: { mode: "preset", preset_id: lightPreset.id },
        strong: { mode: "preset", preset_id: strongPreset.id }
      }
    });
  }

  return (
    <div className="panelBackdrop" role="presentation" onClick={onClose}>
      <aside className="sidePanel" role="dialog" aria-label="运行配置" onClick={(event) => event.stopPropagation()}>
        <div className="sidePanelHeader">

          <button className="iconButton" type="button" aria-label="关闭运行配置" onClick={onClose}>
            ×
          </button>
        </div>
        <div className="sidePanelBody runtimeConfigBody">
          <section className="debugSection">
            <h3>数据库连接</h3>
            <p>{databasePreset.display_name}</p>
          </section>
          <section className="debugSection">
            <h3>模型路由</h3>
            <p>轻量模型：{lightPreset.display_name}</p>
            <p>强模型：{strongPreset.display_name}</p>
```
